# Novelist Phase 1: Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Scaffold a working Tauri v2 + Svelte 5 app where the user can open a directory, browse files in a sidebar, click a file to open it in a CodeMirror 6 editor with basic Markdown syntax highlighting, edit it, and save it back — the core read-edit-write loop.

**Architecture:** Tauri v2 shell with Svelte 5 frontend and Rust backend. Frontend renders a three-column layout (sidebar / editor / outline placeholder). Rust handles all file I/O via Tauri IPC commands with type-safe bindings generated by tauri-specta. CodeMirror 6 provides the editor with Markdown language support.

**Tech Stack:** Tauri v2, Svelte 5 (Runes), Vite 6, Tailwind CSS 4, CodeMirror 6, tauri-specta v2, tokio, serde, toml

**Spec:** `design/design-overview.md`

**Scope:** This is Phase 1 of multiple phases. It covers spec sections 2-4, 5.2, 9.1, 9.3, 12.2 (partial), 13-15. Later phases will add WYSIWYG decorations, file watching, Zen Mode, plugin system, export, themes, etc.

```rust
#[derive(Debug, thiserror::Error)]
pub enum AppError {
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),

    #[error("TOML parse error: {0}")]
    TomlParse(#[from] toml::de::Error),

    #[error("TOML serialize error: {0}")]
    TomlSerialize(#[from] toml::ser::Error),

    #[error("File not found: {0}")]
    FileNotFound(String),

    #[error("Not a directory: {0}")]
    NotADirectory(String),

    #[error("{0}")]
    Custom(String),
}

// Tauri commands need errors to be serializable
impl Serialize for AppError {
    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
    where
        S: serde::Serializer,
    {
        serializer.serialize_str(&self.to_string())
    }
}
```

```rust
use crate::error::AppError;
use crate::models::file_state::FileEntry;
use std::path::Path;

#[tauri::command]
#[specta::specta]
pub async fn read_file(path: String) -> Result<String, AppError> {
    let p = Path::new(&path);
    if !p.exists() {
        return Err(AppError::FileNotFound(path));
    }
    let content = tokio::fs::read_to_string(&path).await?;
    Ok(content)
}

#[tauri::command]
#[specta::specta]
pub async fn write_file(path: String, content: String) -> Result<(), AppError> {
    // Atomic save: write to temp file, then rename
    let temp_path = format!("{}.novelist-tmp", &path);
    tokio::fs::write(&temp_path, &content).await?;
    tokio::fs::rename(&temp_path, &path).await?;
    Ok(())
}

#[tauri::command]
#[specta::specta]
pub async fn list_directory(path: String) -> Result<Vec<FileEntry>, AppError> {
    let p = Path::new(&path);
    if !p.is_dir() {
        return Err(AppError::NotADirectory(path));
    }

    let mut entries = Vec::new();
    let mut read_dir = tokio::fs::read_dir(&path).await?;

    while let Some(entry) = read_dir.next_entry().await? {
        let metadata = entry.metadata().await?;
        let name = entry.file_name().to_string_lossy().to_string();

        // Skip hidden files except .novelist
        if name.starts_with('.') && name != ".novelist" {
            continue;
        }

        entries.push(FileEntry {
            name,
            path: entry.path().to_string_lossy().to_string(),
            is_dir: metadata.is_dir(),
            size: metadata.len(),
        });
    }

    // Sort: directories first, then alphabetical
    entries.sort_by(|a, b| match (a.is_dir, b.is_dir) {
        (true, false) => std::cmp::Ordering::Less,
        (false, true) => std::cmp::Ordering::Greater,
        _ => a.name.to_lowercase().cmp(&b.name.to_lowercase()),
    });

    Ok(entries)
}
```

```rust
use tauri_specta::{collect_commands, Builder};

#[cfg_attr(mobile, tauri::mobile_entry_point)]

            tracing_subscriber::EnvFilter::try_from_default_env()

            read_project_config,

    builder
        .export(
            specta_typescript::Typescript::default()
                .bigint(specta_typescript::BigIntExportBehavior::Number)

    tauri::Builder::default()
        .plugin(tauri_plugin_dialog::init())
        .invoke_handler(builder.invoke_handler())
        .setup(move |app| {
            builder.mount_events(app);
            Ok(())
        })
        .run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

```typescript
/**
 * CJK-aware word count.
 * - CJK characters: each character counts as 1 word
 * - Latin text: standard whitespace-delimited word count
 * - Mixed content: sum of both
 */
export function countWords(text: string): number {
  if (!text.trim()) return 0;

  // Match CJK characters (CJK Unified Ideographs + common ranges)
  const cjkRegex = /[\u4e00-\u9fff\u3400-\u4dbf\uf900-\ufaff\u{20000}-\u{2a6df}\u{2a700}-\u{2b73f}\u{2b740}-\u{2b81f}\u{2b820}-\u{2ceaf}\u{2ceb0}-\u{2ebef}\u{30000}-\u{3134f}\u3000-\u303f\uff00-\uffef]/gu;

  // Count CJK characters
  const cjkMatches = text.match(cjkRegex);
  const cjkCount = cjkMatches ? cjkMatches.length : 0;

  // Remove CJK characters, count remaining Latin words
  const withoutCjk = text.replace(cjkRegex, ' ');
  const latinWords = withoutCjk.split(/\s+/).filter(w => w.length > 0);
  const latinCount = latinWords.length;

  return cjkCount + latinCount;
}
```

```typescript
interface TabState {
  id: string;
  filePath: string;
  fileName: string;
  content: string;
  isDirty: boolean;
  scrollPosition: number;
  cursorPosition: number;
}

class TabsStore {
  tabs = $state<TabState[]>([]);
  activeTabId = $state<string | null>(null);

  get activeTab(): TabState | undefined {
    return this.tabs.find(t => t.id === this.activeTabId);
  }

  openTab(filePath: string, content: string) {
    // If tab already open, just activate it
    const existing = this.tabs.find(t => t.filePath === filePath);
    if (existing) {
      this.activeTabId = existing.id;
      return;
    }

    const fileName = filePath.split('/').pop() || filePath;
    const id = crypto.randomUUID();
    const tab: TabState = {
      id,
      filePath,
      fileName,
      content,
      isDirty: false,
      scrollPosition: 0,
      cursorPosition: 0,
    };
    this.tabs.push(tab);
    this.activeTabId = id;
  }

  closeTab(id: string) {
    const idx = this.tabs.findIndex(t => t.id === id);
    if (idx === -1) return;

    this.tabs.splice(idx, 1);

    // If we closed the active tab, activate the nearest one
    if (this.activeTabId === id) {
      if (this.tabs.length === 0) {
        this.activeTabId = null;
      } else {
        const newIdx = Math.min(idx, this.tabs.length - 1);
        this.activeTabId = this.tabs[newIdx].id;
      }
    }
  }

  activateTab(id: string) {
    this.activeTabId = id;
  }

  updateContent(id: string, content: string) {
    const tab = this.tabs.find(t => t.id === id);
    if (tab) {
      tab.content = content;
      tab.isDirty = true;
    }
  }

  markSaved(id: string) {
    const tab = this.tabs.find(t => t.id === id);
    if (tab) {
      tab.isDirty = false;
    }
  }

  closeAll() {
    this.tabs = [];
    this.activeTabId = null;
  }
}

export const tabsStore = new TabsStore();
```

```svelte
<script lang="ts">
  import { onMount } from 'svelte';
  import { EditorView } from '@codemirror/view';
  import { createEditorExtensions, createEditorState } from '$lib/editor/setup';
  import { tabsStore } from '$lib/stores/tabs.svelte';
  import { commands } from '$lib/ipc/commands';
  import { countWords } from '$lib/utils/wordcount';

  let editorContainer: HTMLDivElement;
  let view: EditorView | null = null;
  let currentTabId: string | null = null;

  // Reactive: track word count and cursor position for status bar
  export let wordCount = $state(0);
  export let cursorLine = $state(1);
  export let cursorCol = $state(1);

  const extensions = createEditorExtensions();

  function setupView(doc: string) {
    if (view) {
      view.destroy();
    }

    const updateListener = EditorView.updateListener.of((update) => {
      if (update.docChanged && tabsStore.activeTab) {
        const content = update.state.doc.toString();
        tabsStore.updateContent(tabsStore.activeTab.id, content);
        wordCount = countWords(content);
      }
      if (update.selectionSet) {
        const pos = update.state.selection.main.head;
        const line = update.state.doc.lineAt(pos);
        cursorLine = line.number;
        cursorCol = pos - line.from + 1;
      }
    });

    const state = createEditorState(doc, [...extensions, updateListener]);
    view = new EditorView({
      state,
      parent: editorContainer,
    });

    wordCount = countWords(doc);
  }

  // React to active tab changes
  $effect(() => {
    const tab = tabsStore.activeTab;
    if (!tab) {
      if (view) {
        view.destroy();
        view = null;
      }
      currentTabId = null;
      return;
    }

    if (tab.id !== currentTabId) {
      currentTabId = tab.id;
      setupView(tab.content);
    }
  });

  // Save with Cmd+S / Ctrl+S
  function handleKeydown(e: KeyboardEvent) {
    if ((e.metaKey || e.ctrlKey) && e.key === 's') {
      e.preventDefault();
      saveCurrentFile();
    }
  }

  async function saveCurrentFile() {
    const tab = tabsStore.activeTab;
    if (!tab || !tab.isDirty) return;

    await commands.writeFile(tab.filePath, tab.content);
    tabsStore.markSaved(tab.id);
  }

  onMount(() => {
    return () => {
      if (view) view.destroy();
    };
  });
</script>
```
